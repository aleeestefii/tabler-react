### Avatars List

```jsx
import { Avatar } from "tabler-react";

<Avatar.List>
  <Avatar imageURL="/demo/faces/male/9.jpg" />
  <Avatar imageURL="/demo/faces/female/8.jpg" />
  <Avatar imageURL="/demo/faces/male/4.jpg" />
</Avatar.List>;
```

### Stacked List

```jsx
import { Avatar } from "tabler-react";

<Avatar.List stacked>
  <Avatar imageURL="/demo/faces/male/9.jpg" />
  <Avatar imageURL="/demo/faces/female/8.jpg" />
  <Avatar imageURL="/demo/faces/male/4.jpg" />
  <Avatar>+8</Avatar>
</Avatar.List>;
```
